partial alphanumeric_keys
xkb_symbols "us" {
    include "us"

    name[Group1]="Swedish (US, with Swedish letters)";

    key <AC10> { [ NoSymbol, NoSymbol, odiaeresis, Odiaeresis ] };
    key <AC11> { [ NoSymbol, NoSymbol, adiaeresis, Adiaeresis ] };
    key <AD11> { [ NoSymbol, NoSymbol, aring, Aring ] };

    include "level3(ralt_switch)"
};

# ÄLDRE DIST ÄN UBUNTU 19.10

Add this to /usr/share/X11/xkb/symbols/se:

partial alphanumeric_keys
xkb_symbols "us" {

    name[Group1]= "Swedish (US, with åäö)";

    include "us(basic)"

    //include "eurosign(5)"

    key <AD11> {	[ aring, Aring, bracketleft,	braceleft	]	};
    key <AC10> {	[ odiaeresis, Odiaeresis, semicolon,	colon		]	};
    key <AC11> {	[ adiaeresis, Adiaeresis, apostrophe,	quotedbl	]	};

    include "level3(ralt_switch)"
};


add the following to /usr/share/X11/xkb/rules/evdev.xml, under the se layout def (search for: <name>se</name>):


        <variant>
          <configItem>
            <name>us</name>
            <description>Swedish (US, with åäö)</description>
          </configItem>
        </variant>


Reload (not needed first time):
sudo rm /var/lib/xkb/*.xkm
sudo dpkg-reconfigure xkb-data
